# python3

import sys

def gcd(a,b):
	while True:
		if a%b == 0:
			return b
		a,b = b,a%b

num = int(sys.stdin.readline())
nlist = [int(x) for x in sys.stdin.readline().split()]
for i in range(1,num):
	nlist[i] = gcd(nlist[i-1],nlist[i])
g = nlist[-1]
for i in range(1,g+1):
	if g//i == g/i:
		print(i)
