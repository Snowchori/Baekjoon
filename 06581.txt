# python3

import sys

sen = sys.stdin.read().split()
for i in range(len(sen)):
	if sen[i] == "<br>":
		print()
	elif sen[i] == "<hr>":
		if i == 0 or i > 1 and sen[i-1] == "<br>" or sen[i-1] == "<hr>":
			print("-"*80)
		else:
			print()
			print("-"*80)
	else:
		print(sen[i],end=" ")
