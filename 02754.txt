# python3

import sys

word = sys.stdin.readline().strip()
if word == "A+":
	print(4.3)
elif word == "A0":
	print(4.0)
elif word == "A-":
	print(3.7)
elif word == "B+":
	print(3.3)
elif word == "B0":
	print(3.0)
elif word == "B-":
	print(2.7)
elif word == "C+":
	print(2.3)
elif word == "C0":
	print(2.0)
elif word == "C-":
	print(1.7)
elif word == "D+":
	print(1.3)
elif word == "D0":
	print(1.0)
elif word == "D-":
	print(0.7)
else:
	print(0.0)
